    批量重命名文件夹

    @echo off
set a=-1
:: 开启变量局部修改
setlocal EnableDelayedExpansion
FOR /f "tokens=*" %%i IN ('dir /a:d /b') DO (
    set /A a+=1
    echo "!a!"
    if /i !a! lss 10 (
        ren "%%i" "P0!a!"
    ) else (
        ren "%%i" "P!a!"
    )
)

    1. 阿里OSS图片上传过程

    阿里OSS图片上传过程

    1.在页面选择图片，使用FileReader读取图片为一个data: URL格式的字符串，再使用Blob转成一个blob对象

    // 上传图片文件处理
function fileHandle(f) {
  let file = f.files[0];
  if (!file) return;
  if (!/^image\/\w+$/.test(file.type)) {
    console.log('图片格式错误')
    return false;
  }
  let reader = new FileReader();
  reader.readAsDataURL(file);
  reader.onloadstart = function() {
    console.log('文件上传处理......')
  };
  // 读取图片文件完成
  reader.onload = function(e) {
    let blob = new Blob([e.target.result], {
      type: file.type
    });
    let params = {
      blob: blob,
      name: file.name,
    };
    upLoad(params, (call) => {
      console.log(call);
    })
  };
}

    2.引用阿里oss-js-sdk，使用OSS.urllib.request向我们的服务器发送请求(安全)，获取配置参数(AccessKeyId,AccessKeySecret,SecurityToken)，使用put添加对象到bucket，再将图片上传到服务器。

    // oss上传图片到阿里云
upLoad(e, callback) {
  // 服务器地址
  let url = "https://www.test.com/";
  // Vue引入sdk let OSS = require("ali-oss");
  OSS.urllib.request(url + "resource/aliyun/sts", {
    method: "get"
  }).then(result => {
    let creds = JSON.parse(result.data);
    if (creds.StatusCode == 200) {
      let client = new OSS({
        region: "oss-cn-shanghai",
        accessKeyId: creds.AccessKeyId,
        accessKeySecret: creds.AccessKeySecret,
        stsToken: creds.SecurityToken,
        bucket: "img-syd"
      });
      let fileName = e.name.substr(0, e.name.lastIndexOf("."));
      client.put(fileName, e.blob).then(res => {
        let upPath;
      if(e.blob.type.indexOf("image")){
        upPath = 'resource/aliyun/videoUrl';
      } else {
        upPath = 'resource/aliyun/fileUrl';
      }
      OSS.urllib.request( url + upPath + "?filename=" + fileName, {
        method: "POST"
      }).then(result => {
        callback(JSON.parse(result.data));
      });
    });
    }
  });
}
